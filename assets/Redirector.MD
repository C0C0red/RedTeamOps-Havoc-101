
```

RewriteCond %{REQUEST_URI} "demon.bin" [NC]
RewriteRule ^.*$ "https://localhost:9090%{REQUEST_URI}" [P]

RewriteRule ^.*$ "https://localhost:8443%{REQUEST_URI}" [P]

```

```

```
